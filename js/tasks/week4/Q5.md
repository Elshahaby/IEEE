## Q5

```js
function* FibonacciGenerator() {
  let tmp = 0;
  let tmp2 = 1;

  yield tmp;
  yield tmp2;

  while (true) {
    let next = tmp + tmp2;
    yield next;
    tmp = tmp2;
    tmp2 = next;
  }
}

let fibo = FibonacciGenerator();

for (let i = 0; i < 20; i++) {
  console.log(fibo.next().value);
}
```
